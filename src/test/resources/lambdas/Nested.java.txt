package lambdas;

import java.util.function.Function;

public class Nested {
	public void run() {
		Function<Integer, Integer> x1 = y -> test(z->z);
		Function<Integer, Integer> x2 = y -> this.test(z->z);
	}

	public Integer test (Function<Integer, Integer> another) {
		return another.apply(1);
	}
}
